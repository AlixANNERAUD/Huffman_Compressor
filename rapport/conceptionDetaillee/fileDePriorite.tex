        \champEnregistrement{tab}{\tableau{1..MAX}{of}{\huffmanTree}}

    \fonction{priorityQueue}{}{\priorityQueue}{}
    {
        pQ : \priorityQueue
    }{
        \affecter{\champ{pQ}{length}}{0}
        \retourner{pQ}
        }
    
    \fonction{get\_length}{pQ : \priorityQueue}{\naturel}{}{}{\retourner{\champ{pQ}{length}}}
    
    \fonction{pop}{pQ : \priorityQueue, }{\priorityQueue, Element}{get\_length(pQ) > 0}{}{
        \affecter{\champ{pQ}{length}}{\champ{pQ}{length}-1}
        \retourner{\champ{pQ}{tab}[\champ{pQ}{length}+1]}}

    \fonction{push}{pQ : \priorityQueue,  e : Element}{\priorityQueue}{get\_length(pQ) < MAXLENGTH}{temp : \huffmanTree}
    {
        \affecter{\champ{pQ}{length}}{\champ{pQ}{length}+1}
        \sialors{\champ{e}{weight} = \champ{\champ{pQ}{tab}[1]}{weight}}{
            \sialors {\champ{e}{value} <= \champ{\champ{pQ}{tab}[1]}{value}}{
                    \affecter{temp}{\champ{pQ}{tab}[1]}
                    \affecter{\champ{pQ}{tab}[1]}{e}
                    \affecter{e}{\champ{pQ}{tab}[1]}
                }
        }
        \pour{i}{1}{\champ{pQ}{length}}{}{
        \sialors{\champ{e}{weight}>\champ{\champ{pQ}{tab}[i]}{weight} et \champ{e}{weight} <= \champ{\champ{pQ}{tab}[i+1]}{weight}}{
            \sialorssinon{\champ{e}{weight} = \champ{\champ{pQ}{tab}[i+1]}{weight}}{
                \sialors {\champ{e}{value} <= \champ{\champ{pQ}{tab}[i+1]}{value}}{
                    \affecter{temp}{\champ{pQ}{tab}[i+1]}
                    \affecter{\champ{pQ}{tab}[i+1]}{e}
                    \affecter{e}{temp}
                }
            }
            {
                \affecter{temp}{\champ{pQ}{tab}[i+1]}
                \affecter{\champ{pQ}{tab}[i+1]}{e}
                \affecter{e}{temp}
            }
        }}
        \retourner{pQ}
    }

\end{algorithme}
