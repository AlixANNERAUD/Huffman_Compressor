\section{Statistiques}

\begin{algorithme}
    \begin{enregistrement}{statistics}
        \champEnregistrement{lesElements}{\tableau{1..256}{d'}{Element}}
        \champEnregistrement{nbOccurrence}{\tableau{1..256}{de}{\naturel}}
    \end{enregistrement}
    \espace
    \fonction{statistics}{}{\statistics}{}
    {res : \statistics, i : \naturel}
    {
        \affecter{\champ{res}{lesElements}}{\tableau{1..256}{d'}{Element}}
        \affecter{\champ{res}{nbOccurrence}}{\tableau{1..256}{de}{\naturel}}
        \pour{i}{1}{256}{}{
            \affecter{\champ{res}{nbOccurence}[i]}{0}
        }
        \retourner{res}        
    }
    \espace
    \fonction{get\_count}{stat : \statistics, e : Element}{\naturel}{}
    {i : \naturel}
    {
        \affecter{i}{1}
        \tantque{i $\leq$ 255 et non({e = {\champ{stat}{lesElements}}[i]})}{
            \affecter{i}{i+1}
        }
        \retourner{{\champ{stat}{nbOccurrence}}[i]}    
    }
    \espace
    \procedure{set\_count}{\paramEntreeSortie{stat : \statistics}, \paramEntree{e : Element, n : \naturel}}{}
    {i : \naturel}
    {
        \affecter{i}{1}
        \tantque{i $\leq$ 255 et non({e = {\champ{stat}{lesElements}}[i]})}{
            \affecter{i}{i+1}
        }
        \affecter{{\champ{stat}{nbOccurrence}}[i]}{n}
    }
    \espace
    \procedure{increase\_count}{\paramEntreeSortie{stat : \statistics}, \paramEntree{e : Element}}{}
    {i : \naturel}
    {
        \affecter{i}{1}
        \tantque{i $\leq$ 255 et non({e = {\champ{stat}{lesElements}}[i]})}{
            \affecter{i}{i+1}
        }
        \affecter{{\champ{stat}{nbOccurrence}}[i]}{{\champ{stat}{nbOccurrence}}[i]+1}
    }
    \espace
    \fonction{get\_total\_count}{stat : \statistics}{\naturel}{}
    {sum, i : \naturel}
    {
        \affecter{sum}{0}
        \pour{i}{1}{256}{}{
            \affecter{sum}{sum + \champ{stat}{nbOccurence}[i]}
        }
        \retourner{sum}
    }
    \espace
    \fonction{statistics\_from\_file}{file : \binaryFile}{\statistics}{}
    {stat : \statistics, e : Element}
    {
        \affecter{stat}{\instruction{statistics()}}
        \tantque{non(\instruction{is\_end(file)})}{
            \affecter{e}{\instruction{readByte(file)}}
            \instruction{increase\_count(stat, e)}
        }
      
        \retourner{stat}
    }
    \espace
    \fonction{are\_equal}{s1, s2 : \statistics}{\booleen}{}
    {eq : \booleen, i : \naturel}
    {
        \affecter{eq}{Vrai}
        \tantque{eq et i < 256}{
            \sialors{non((\champ{s1}{lesElements}[i] = \champ{s2}{lesElements}[i]) et (\champ{s1}{nbOccurence}[i] = \champ{s2}{nbOccurence}[i]))}{
                \affecter{eq}{Faux}
            }
        }
        \retourner{eq}
    }
    \espace
\end{algorithme}
