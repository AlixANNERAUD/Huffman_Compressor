\section{Arbre de Huffman}


\begin{algorithme}

    \begin {enregistrement}{Node}
        \champEnregistrement{value}{\byte}
        \champEnregistrement{weight}{\naturel}
        \champEnregistrement{leftChild}{\huffmanTree}
        \champEnregistrement{rightChild}{\huffmanTree}
    \end{enregistrement}
    \espace
    \type {HuffmanTree}{\pointeur {Node}}
    \espace
    \fonction{new\_leaf}{value : \byte, weight : \naturel}{\huffmanTree}{}
    {ht : huffmanTree}
    {
       \affecter{\pointeur{ht}.value}{value}
       \affecter{\pointeur{ht}.weight}{weight}
       \affecter{\pointeur{\pointeur{ht}.leftChild}}{\pointeurNULL}
       \affecter{\pointeur{\pointeur{ht}.rightChild}}{\pointeurNULL}
       \retourner{ht}
    }
    \espace
    \fonction{new\_tree}{tree1 : \huffmanTree,tree2 : \huffmanTree}{\huffmanTree}{}
    {ht : huffmanTree}
    {
        \affecter {\pointeur{ht}} {new\_leaf(byte(0), \pointeur{tree1}.weight + \pointeur{tree2}.weight)}
        \affecter {\pointeur{\pointeur{ht}.leftChild}}{tree1}
        \affecter {\pointeur{\pointeur{ht}.rightChild}}{tree2}
    }
    \espace
    \fonction{is\_leaf}{tree : \huffmanTree}{\booleen}{}
    {}
    {
        \retourner {(\pointeur{tree}.leftChild = \pointeurNULL) et (\pointeur{tree}.rightChild = \pointeurNULL)}
    }
    \espace
    \fonction{get\_value}{tree : \huffmanTree}{\byte}{is\_leaf(tree)}
    {}
    {
        \retourner {\pointeur{tree}.value}
    }
    \espace
    \fonction{get\_weight}{tree : \huffmanTree}{\naturel}{}
    {}
    {
        \retourner {\pointeur{tree}.weight}
    }
    \espace
    \fonction{get\_left\_child}{tree : \huffmanTree}{\typePointeur{\huffmanTree}}{non(is\_leaf(tree))}
    {}
    {
        \retourner {\pointeur{\pointeur{tree}.leftChild}}
    }
    \espace
    \fonction{get\_right\_child}{tree : \huffmanTree}{\typePointeur{\huffmanTree}}{non(is\_leaf(tree))}
    {}
    {
        \retourner {\pointeur{\pointeur{tree}.rightChild}}
    }
    \espace
    \fonction{is\_greater\_than}{tree1, tree2 : \huffmanTree}{\booleen}{}
    {}
    {
        \sialors{get\_weight(tree1) > get\_weight(tree2)}{
            \retourner{Vrai}
        }
        \sialors{get\_weight(tree1) = get\_weight(tree2) et is\_leaf(tree1) et is\_leaf(tree2)}{
            \retourner{get\_value(tree1) > get\_value(tree2)}
        }

        \retourner{Faux}
    }
    \espace
\end{algorithme}
