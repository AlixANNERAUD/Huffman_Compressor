\section{Octet}

\newcommand{\byte}{\textbf{Byte}}

\begin{tad}
    \tadNom{\byte}
    \tadParametres{Bit}
    \tadDependances{Bit, Natural}
    \begin{tadOperations}{obtenirElement}
        \tadOperation{byte}{}{\tadParams{\byte}}
        \tadOperationAvecPreconditions{changeibit}{\tadParams{\byte,Natural}}{\tadParams{\byte}}
        \tadOperationAvecPreconditions{getbit}{\tadParams{\byte, Natural}}{\tadParams{\bit}}
    \end{tadOperations}
    \begin{tadSemantiques}{\byte}
        \tadSemantique{Byte}{crée le Byte 0.}
        \tadSemantique{changeibit}{change le ième bit.}
        \tadSemantique{getbit}{retourne le ième bit.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{getbit(a,i)=getbit(changeibit(changeibit(a,i),i),i)}
    \end{tadAxiomes}
    \begin{tadPreconditions}{TableDeCodage}
		\tadPrecondition{changeibit(a,i)}{1 $\leq$ i $\leq$ 8}
		\tadPrecondition{getbit(a,i)}{1 $\leq$ i $\leq$ 8}
    \end{tadPreconditions}
\end{tad}
