\subsection{Byte}

\begin{tad}
    \tadNom{Byte}
    \tadParametres{Bit}
    \tadDependances{Bit, Natural}
    \begin{tadOperations}{obtenirElement}
        \tadOperation{byte}{}{\tadParams{Byte}}
        \tadOperationAvecPreconditions{changeibit}{\tadParams{Byte,Natural}}{\tadParams{Byte}}
        \tadOperationAvecPreconditions{getibit}{\tadParams{Byte, Natural}}{\tadParams{Bit}}
    \end{tadOperations}
    \begin{tadSemantiques}{Byte}
        \tadSemantique{Byte}{create a Byte 0.}
        \tadSemantique{changeibit}{change the ith bit.}
        \tadSemantique{getibit}{return the ith bit.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{getibit(a,i)=getibit(changeibit(changeibit(a,i),i),i)}
    \end{tadAxiomes}
    \begin{tadPreconditions}{TableDeCodage}
		\tadPrecondition{changeibit(a,i)}{1 $\leq$ i $\leq$ 8}
		\tadPrecondition{getibit(a,i)}{1 $\leq$ i $\leq$ 8}
    \end{tadPreconditions}
\end{tad}
