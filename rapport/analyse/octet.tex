\section{Octet}

\newcommand{\byte}{\textbf{Byte}}

\begin{tad}
    \tadNom{\byte}
    \tadDependances{\bit, \naturel}
    \begin{tadOperations}{getBit}
        \tadOperationAvecPreconditions{byte}{\naturel}{\byte}
        \tadOperationAvecPreconditions{setBit}{\tadParams{\byte,\naturel,\bit}}{\byte}
        \tadOperationAvecPreconditions{getBit}{\tadParams{\byte, \naturel}}{\bit}
    \end{tadOperations}
    \begin{tadSemantiques}{getBit}
        \tadSemantique{byte}{Crée un octet a partir d'un naturel.}
        \tadSemantique{setBit}{Défini le i-ème bit.}
        \tadSemantique{getBit}{Obtient le i-ème bit.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{getBit(setBit(byte, i, bit), i) = bit}
    \end{tadAxiomes}
    \begin{tadPreconditions}{getBit(byte, i)}
        \tadPrecondition{byte(natural)}{natural < 256}
		\tadPrecondition{setBit(byte, i)}{1 $\leq$ i $\leq$ 8}
		\tadPrecondition{getBit(byte, i)}{1 $\leq$ i $\leq$ 8}
    \end{tadPreconditions}
\end{tad}
