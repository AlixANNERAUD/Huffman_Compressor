\section{Arbre d'Huffman}

\newcommand{\huffmanTree}{\textbf{HuffmanTree}}

\begin{tad}
    \tadNom{\huffmanTree}
    \tadParametres{Element}
    \tadDependances{\naturel, \huffmanTree}
    \begin{tadOperations}{getLeftChild}
        \tadOperation{newLeaf}{\tadParams{Element,\naturel}}{\huffmanTree}
        \tadOperation{newTree}{\tadParams{\huffmanTree, \huffmanTree}}{\huffmanTree}
        \tadOperation{isLeaf}{\huffmanTree}{\booleen}
        \tadOperationAvecPreconditions{getValue}{\huffmanTree}{Element}
        \tadOperation{getWeight}{\huffmanTree}{\naturel}
        \tadOperationAvecPreconditions{getRightChild}{\huffmanTree}{\huffmanTree}
        \tadOperationAvecPreconditions{getLeftChild}{\huffmanTree}{\huffmanTree}
        \tadOperation{isGreaterThan}{\huffmanTree, \huffmanTree}{\booleen}
    \end{tadOperations}
    \begin{tadSemantiques}{getLeftChild}
        \tadSemantique{newLeaf}{Créer un nœud d'Huffman de type feuille (avec un poids et une valeur, mais sans fils).}
        \tadSemantique{newTree}{Créer un nœud d'Huffman de type arbre (avec un poids et deux fils, mais sans valeur).}
        \tadSemantique{isLeaf}{Renvoi si le noeud est une feuille.}
        \tadSemantique{getValue}{Renvoi la valeur du noeud.}
        \tadSemantique{getRightChild}{Renvoi le fils droit du noeud.}
        \tadSemantique{getLeftChild}{Renvoi le fils gauche du noeud.}
        \tadSemantique{isGreaterThan}{Renvoi vrai si le poids du premier arbre est supérieur à celui du second, ou dans le cas où ce sont des feuilles de même poids, si sa valeur est supérieure. Faux sinon.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{isLeaf(newTree(newLeaf(octet(16), 4), newLeaf(octet(32), 2))) = faux}
        \tadAxiome{isLeaf(newLeaf(byte(16), 4)) = vrai}
        \tadAxiome{getWeight(newTree(newLeaf(byte(16), 4), newLeaf(byte(32), 2))) = 6} 
        \tadAxiome{getValue(newLeaf(byte(16), 4)) = byte(16)}
    \end{tadAxiomes}
    \begin{tadPreconditions}{getLeftChild(tree)}
        \tadPrecondition{getValue(tree)}{isLeaf(tree)}
        \tadPrecondition{getRightChild(tree)}{non(isLeaf(tree))}
        \tadPrecondition{getLeftChild(tree)}{non(isLeaf(tree))}
    \end{tadPreconditions}
\end{tad}
