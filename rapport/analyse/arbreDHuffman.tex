\section{Arbre d'Huffman}

\newcommand{\huffmanTree}{\textbf{HuffmanTree}}

\begin{tad}
    \tadNom{\huffmanTree}
    \tadParametres{\byte, \naturel, \booleen, \typePointeur{\huffmanTree}}
    \tadDependances{\byte, \naturel, \huffmanTree}
    \begin{tadOperations}{setLeaf}
        \tadOperation{huffmanTree}{\tadParams{\naturel}}{\huffmanTree}
        \tadOperation{delete}{\tadParams{\huffmanTree}}{\huffmanTree}
        
        \tadOperation{isValid}{\huffmanTree}{\booleen}
        \tadOperation{hasValue}{\huffmanTree}{\booleen}
        \tadOperationAvecPreconditions{getValue}{\huffmanTree}{\byte}
        \tadOperation{getWeight}{\huffmanTree}{\naturel}
        \tadOperation{getDepth}{\huffmanTree}{\naturel}
        \tadOperationAvecPreconditions{getRightLeaf}{\huffmanTree}{\huffmanTree}
        \tadOperationAvecPreconditions{getLeftLeaf}{\huffmanTree}{\huffmanTree}
    
        \tadOperation{setValue}{\huffmanTree}{\huffmanTree}
        \tadOperation{setLeaf}{\tadParams{\huffmanTree, \huffmanTree, \huffmanTree}}{\huffmanTree}
    \end{tadOperations}
    \begin{tadSemantiques}{getDepth}
        \tadSemantique{huffmanTree}{Crée une arbre d'huffman.}
        \tadSemantique{delete}{Détruit l'arbre ainsi que tout ses fils de manière récursive.}
        \tadSemantique{getValue}{Renvoi la valeur du noeud.}
        \tadSemantique{getWeight}{Renvoi le poids du noeud.}
        \tadSemantique{getDepth}{Renvoie la profondeur de l'arbre.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{getDepth(huffmanTree('A', 4)) = 1}
        \tadAxiome{getWeight(huffmanTree(octet(16), 4)) = octet(16)}
        \tadAxiome{getValue(huffmanTree(octet(16), 4)) = octet(16)}
    \end{tadAxiomes}
    \begin{tadPreconditions}{getLeftLeaf(tree)}
        \tadPrecondition{isValid(tree)}{getDepth(tree) > 0}
        \tadPrecondition{getValue(tree)}{hasValue(tree)}
        \tadPrecondition{getRightLeaf(tree)}{getDepth(tree) > 0}
        \tadPrecondition{getLeftLeaf(tree)}{getDepth(tree) > 0}
    \end{tadPreconditions}
\end{tad}
