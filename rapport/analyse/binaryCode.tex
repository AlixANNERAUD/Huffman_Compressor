\section{Code binaire}

\newcommand{\binaryCode}{\textbf{BinaryCode}}

\begin{tad}
    \tadNom{\binaryCode}
    \tadParametres{}
    \tadDependances{\bit, \naturel}
    \begin{tadOperations}{clone}
        \tadOperation{binaryCode}{}{\binaryCode}
        \tadOperation{getLength}{\tadParams{\binaryCode}}{\naturel}
        \tadOperationAvecPreconditions{getBit}{\tadParams{\binaryCode, \naturel}}{\bit}
        \tadOperationAvecPreconditions{addBit}{\tadParams{\binaryCode, \bit}}{\binaryCode}
        \tadOperationAvecPreconditions{deleteBit}{\tadParams{\binaryCode, \naturel}}{\binaryCode}
        \tadOperationAvecPreconditions{deleteLastBit}{\binaryCode}{\binaryCode}
        \tadOperation{clone}{\binaryCode}{\binaryCode}
    \end{tadOperations}
    \begin{tadSemantiques}{clone}
        \tadSemantique{binaryCode}{Crée un code binaire vide.}
        \tadSemantique{getLength}{Renvoie la longueur d'un code binaire (le nombre de bits).}
        \tadSemantique{getBit}{Renvoie un bit d'une position donnée d'un code binaire.}
        \tadSemantique{addBit}{Ajoute un bit à la fin d'un code binaire.}
        \tadSemantique{deleteBit}{Supprime un bit d'une position donnée d'un code binaire.}
        \tadSemantique{deleteLastBit}{Supprime le dernier bit d'un code binaire.}
        \tadSemantique{clone}{Crée une copie d'un code binaire.}
    \end{tadSemantiques}
    \begin{tadAxiomes}
        \tadAxiome{getLength(code)=getLength(clone(code))}
        \tadAxiome{getBit(code,i)=getBit(clone(code),i)}
        \tadAxiome{getLength(code)=getLength(deleteLastBit(code))+1}
        \tadAxiome{getBit(code,getLength(code))=getBit(deleteLastBit(addBit(code,bit)),getLength(code))}
        \tadAxiome{getBit(addBit(code,bit),getLength(code))=bit}
    \end{tadAxiomes}
    \begin{tadPreconditions}{deleteLastBit(code)}
		\tadPrecondition{getBit(code, i)}{0 $\leqslant$ i $\leqslant$ getLength(code)}
        \tadPrecondition{addBit(code, bit)}{getLength(code) < MAXIMUM\_LENGTH}
        \tadPrecondition{deleteBit(code, i)}{0 $\leqslant$ i $\leqslant$ getLength(code)}
        \tadPrecondition{deleteLastBit(code)}{getLength(code) $\ne$ 0}
    \end{tadPreconditions}
\end{tad}
