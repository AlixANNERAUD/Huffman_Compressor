\section{Table De Codage}

\newcommand{\codingTable}{\textbf{CodingTable}}

\begin{tad}
  \tadNom{\codingTable}
  \tadDependances{\byte, Naturel, \binaryCode, \huffmanTree}
  \begin{tadOperations}{tableLength}
    \tadOperation{codingTable}{}{\tadParams{\codingTable}}
    \tadOperationAvecPreconditions{fillTable}{\tadParams{\huffmanTree, Naturel, \codingTable}}{\tadParams{\codingTable}}
    \tadOperationAvecPreconditions{addTable}{\tadParams{\codingTable, \byte, \binaryCode}}{\tadParams{\codingTable}}
    \tadOperationAvecPreconditions{getTableCode}{\tadParams{\codingTable, \byte}}{\tadParams{\binaryCode}}
    \tadOperation{isPresent}{\tadParams{\codingTable, \byte}}{\tadParams{Booléen}}
    \tadOperation{tableLength}{\tadParams{\codingTable}}{\tadParams{Naturel}}
  \end{tadOperations}
  \begin{tadSemantiques}{tableLength}
    \tadSemantique{codingTable}{crée une Table de Codage Vide}
    \tadSemantique{fillTable}{rempli une table de codage à partir d'un Arbre de Huffman}
    \tadSemantique{addTable}{ajoute un élément(clé Octet + fréquence Code Binaire) à la table}
    \tadSemantique{getTableCode}{retourne le Code Binaire associé à un Octet donné dans la Table de Codage}
    \tadSemantique{isPresent}{verifie la présence d'une entrée de la table}
    \tadSemantique{tableLength}{calcul le nombre d'entrées dans la table}
  \end{tadSemantiques}
  \begin{tadAxiomes}
    \tadAxiome{getTableCode(addTable(Table,code,oct),oct)=code}
    \tadAxiome{isPresent(addTable(Table,code,oct),oct)}
    \tadAxiome{non(isPresent(Table(),oct))}
    \tadAxiome{tableLength(Table())=0}
    \tadAxiome{tableLength(addTable(Table(),code,oct))=1}
  \end{tadAxiomes}
  \begin{tadPreconditions}{fillTable(Table,arbre)}
    \tadPrecondition{addTable(Table,code,oct)}{non(isPresent(Table,code))}
    \tadPrecondition{getTableCode(Table,code)}{isPresent(Table,code)}
    \tadPrecondition{fillTable(Table,arbre)}{tableLength(Table) = 0}
  \end{tadPreconditions}
\end{tad}
