\subsection{Table De Codage}



\newcommand{\codingTable}{\textbf{CodingTable}}

\begin{tad}
  \tadNom{\codingTable}
  \tadDependances{\byte, \naturel, \binaryCode, \booleen}
  \begin{tadOperations}{coding\_table}
    \tadOperation{coding\_table}{}{\tadParams{\codingTable}}
    \tadOperationAvecPreconditions{add}{\tadParams{\codingTable, \byte, \binaryCode}}{\codingTable}
    \tadOperationAvecPreconditions{get\_value}{\tadParams{\codingTable, \byte}}{\binaryCode}
    \tadOperation{search}{\tadParams{\codingTable, \byte, \naturel}}{\booleen}
  \end{tadOperations}
  \begin{tadSemantiques}{coding\_table}
    \tadSemantique{coding\_table}{Crée une table de codage vide.}
    \tadSemantique{add}{Ajoute une entrée clé (octet) / valeur (code binaire) à la table.}
    \tadSemantique{get\_value}{Retourne le code binaire associé à un octet donné dans la table de codage.}
    \tadSemantique{search}{Retourne la présence d'une clé (octet) dans la table et l'indice de la clé si besoin.}
  \end{tadSemantiques}
  \begin{tadAxiomes}
    \tadAxiome{get\_value(add(Table,oct,code),oct)=code}
    \tadAxiome{search(add(Table,oct,code),oct,i)}
    \tadAxiome{non(search(codingTable(),oct),i)}
  \end{tadAxiomes}
  \begin{tadPreconditions}{get\_value(table, byte)}
    \tadPrecondition{add(table, byte, binary\_code)}{non(search(table, byte),i)}
    \tadPrecondition{get\_value(table, byte)}{search(table, byte,i)}
  \end{tadPreconditions}
\end{tad}
