\section{Statistics}

\newcommand{\statistics}{\textbf{Statistics}}

\begin{tad}
	\tadNom{\statistics}
	\tadParametres{Element}
	\tadDependances{\naturel}
	\begin{tadOperations}{insert}
		\tadOperation{statistics}{}{\tadParams{\statistics}}
		\tadOperation{getCount}{\tadParams{\statistics, Element}}{\naturel}
		\tadOperation{setCount}{\tadParams{\statistics, Element, \naturel}}{\tadParams{\statistics}}
		\tadOperation{increaseCount}{\tadParams{\statistics, Element}}{\tadParams{\statistics}}
	\end{tadOperations}
	\begin{tadSemantiques}{increaseCount}
		\tadSemantique{statistics}{Crée un tableau de Statistics à 0 pour chaque element.}
		\tadSemantique{getCount}{Renvoi le nombre d'occurences d'un element.}
		\tadSemantique{setCount}{Permet d'assigner un nombre d'occurences à un element.}
		\tadSemantique{increaseCount}{Permet d'augmenter d'un l'occurence d'un élément.}
	\end{tadSemantiques}
	\begin{tadAxiomes}
		\tadAxiome{getCount(element, statistics()) = 0 \forall element}
		\tadAxiome{getCount(element, setCount(byte, statistics(), x)) = x}
	\end{tadAxiomes}
\end{tad}
